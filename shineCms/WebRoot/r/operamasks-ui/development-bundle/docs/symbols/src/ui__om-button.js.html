 15 ;(function($) {
 16 	/**
 17      * @name omButton
 18      * @class 按钮组件。类似于html中的button、input[type=submit]、input[type=button]，使用背景图片实现圆角，支持icon，可以只显示icon。<br/><br/>
 19      * <b>特点：</b><br/>
 20      * <ol>
 21      *      <li>实现圆角</li>
 22      *      <li>支持左icon和右icon，可同时出现左右icon，也可以只显示icon不显示label</li>
 23      *      <li>按钮文字数目不限，也可以任意设置按钮宽度</li>
 24      *      <li>支持input[type=button]、input[type=submit]、input[type=reset]、button、a五种标签形式</li>
 25      * </ol>
 26      * 
 27      * <b>示例：</b><br/>
 28      * <pre>
 29      * <script type="text/javascript" >
 30      * $(document).ready(function() {
 31      *     $('#bnt').omButton({
 32      *         icons : {left:'images/help.png',right:'images/edit_add.png'},
 33      *         width : 150,
 34      *         disabled : 'disabled',
 35      *         onClick : function(event){
 36      *             // do something
 37      *         }
 38      *     });
 39      * });
 40      * </script>
 41      * 
 42      * <input id="btn" type="submit" />
 43      * </pre>
 44      * 
 45      * @constructor
 46      * @description 构造函数. 
 47      * @param p 标准config对象：{}
 48      */
 49     $.widget('om.omButton', {
 50     	
 51         options: /**@lends omButton# */{
 52         	/**
 53         	 * 是否禁用组件。可通过('#id').attr('disabled')判断按钮是否禁用。
 54              * @type String 
 55              * @default null 
 56              * @example
 57              * $("#button").omButton({disabled: true}); 
 58              */
 59             disabled : null ,
 60             /**
 61              * 显示文本。label值可以写到dom元素里面，也可以设置在属性里面，设置到属性里面的优先级最高。
 62              * @type String 
 63              * @default ""
 64              * @example
 65              * $("#button").omButton({label: "apusic"});
 66              */
 67             label : null ,
 68             /**
 69         	 * 显示按钮图标，left表示左图标，right表示右图标，取值均为图片路径。
 70              * @type Object 
 71              * @default null 
 72              * @example
 73              * $("#button").omButton({
 74              *     icons: {
 75              *         left: 'images/help.png',
 76              *         right: 'images/edit_add.png'
 77              *     }
 78              * });
 79              */
 80             icons: {
 81     			left: null,
 82     			right: null
 83     		},
 84     		/**
 85         	 * 按钮宽度，设置固定宽度之后按钮将不会随文字的多少而改变。
 86              * @type Number 
 87              * @default null 
 88              * @example
 89              * width : 150
 90              */
 91             width : null ,
 92             /**
 93         	 * 点击按钮时触发的事件。
 94              * @event
 95              * @name omButton#onClick 
 96              * @example
 97              * onClick : function(){
 98              *    //do something
 99              * }
100              */
101             onClick : null
102         },
103         _create:function(){
104             var self = this,
105                 options = this.options;
106             if ( typeof options.disabled != "boolean" ) {
107                 options.disabled = this.element.propAttr( "disabled" );
108     		}
109             this._determineButtonType();
110             if ( this.buttonElement.attr('disabled') == 'disabled' || options.disabled == 'disabled') {
111     			options.disabled = true;
112     		}
113             this._initButton();
114             if(options.disabled){
115             	self._addClass('disabled');
116             	this.buttonElement.css('cursor','default');
117             }
118             var newButtonElement = this.buttonElement.parent().parent();
119             newButtonElement.bind( 'mouseenter.button',function( event ){
120             					if ( options.disabled ) {
121             						return false;
122             					}
123             					self._addClass('hover');
124             				}).bind( "mouseleave.button", function( event ) {
125             					if ( options.disabled ) {
126             						return false;
127             					}
128             					self._removeClass('hover');
129             					self._removeClass('active');
130             				}).bind( "click.button", function( event ){
131             					if ( options.disabled ) {
132             						event.preventDefault();
133             						event.stopImmediatePropagation();
134             						return false;
135             					}else if(self.options.onClick){
136             					    self.options.onClick();
137             					}
138             				}).bind( "mousedown.button", function( event ) {
139             					if ( options.disabled ) {
140             						return false;
141             					}
142             					self._addClass('active');
143             				    self._removeClass('focus');
144             					var onClick = options.onClick;
145         		                if ( onClick && onClick( event ) === false ) {
146         		                    return;
147         		                }
148             				})
149             				.bind( "mouseup.button", function( event ) {
150             					if ( options.disabled ) {
151             						return false;
152             					}
153             					self._addClass('focus');
154             					self._removeClass('active');
155             				})
156             				.bind( "keydown.button", function(event) {
157             					if ( options.disabled ) {
158             						return false;
159             					}
160             					if ( event.keyCode == $.ui.keyCode.SPACE || event.keyCode == $.ui.keyCode.ENTER ) {
161             						self._addClass('active');
162             					}
163             					if( event.keyCode == $.ui.keyCode.SPACE){
164             						var onClick = options.onClick;
165             		                if ( onClick && onClick( event ) === false ) {
166             		                    return;
167             		                }
168             					}
169             				})
170             				.bind( "keyup.button", function() {
171             					self._removeClass('active');
172             				});
173 	            this.element.bind( "focus", function( event ) {
174 								if ( options.disabled ) {
175 									return false;
176 								}
177 								self._addClass('focus');
178 							}).bind( "blur", function( event ) {
179 	        					if ( options.disabled ) {
180 	        						return false;
181 	        					}
182 	        					self._removeClass('focus');
183 	        				});
184         },
185         /**
186          * 启用组件。
187          * @name omButton#enable
188          * @function
189          * @example
190          * $('#btn').omButton('enable');
191          */
192         enable : function(){
193             this._removeClass('disabled');
194             this.options.disabled = false;
195             this.buttonElement.css('cursor','pointer');
196             return this.element.removeAttr('disabled');
197         },
198         /**
199          * 禁用组件。
200          * @name omButton#disable
201          * @function
202          * @example
203          * $('#btn').omButton('disable');
204          */
205         disable : function(){
206         	this._addClass('disabled');
207             this.options.disabled = true;
208             this.buttonElement.css('cursor','default');
209             return this.element.attr('disabled', 'disabled');
210         },
211         /**
212          * 触发点击事件。
213          * @name omButton#click
214          * @function
215          * @example
216          * $('#btn').omButton('click');
217          */
218         click : function(){
219         	if(!this.options.disabled && this.options.onClick){
220         	    this.options.onClick();
221             }
222         },
223         /**
224          * 改变按钮的label属性。
225          * @name omButton#changeLabel
226          * @function
227          * @param label 按钮文本
228          * @example
229          * $('#btn').omButton('changeLabel','按钮label');
230          */
231         changeLabel : function(label){
232             if(this.type == 'a'){
233             	this.buttonElement.text(label) ;
234             }else if( this.type == 'input' ){
235             	this.buttonElement.val(label) ;
236             }else if ( this.type == 'button' ){
237             	this.buttonElement.html(label) ;
238             }
239         },
240         /**
241          * 改变按钮的icon属性。
242          * @name omButton#changeIcons
243          * @function
244          * @param icons 图标路径
245          * @example
246          * $('#btn').omButton('changeIcons',{
247          *     left: 'images/help.png',
248          *     right: 'images/edit_add.png'
249          * });
250          */
251         changeIcons : function( icons ){
252         	if( !this.options.disabled ){
253 	            if( icons ){
254 	            	icons.left?this.buttonElement.css( 'backgroundImage','url( '+icons.left+' )' ):null;
255 	            	icons.right?this.buttonElement.next().attr( 'src',icons.right ):null;
256 	            }
257             }
258         },
259         _addClass : function( state ){
260         	this.buttonElement.parent().parent().addClass( 'om-state-'+state );
261         },
262         _removeClass : function( state ){
263         	this.buttonElement.parent().parent().removeClass( 'om-state-'+state );
264         },
265         _initButton : function(){
266         	var options = this.options;
267         	var label = this._getLabel();
268         	var wrapperSpan = $( '<span>' ).addClass( 'om-btn om-state-default').css('border','none'),
269         	    leftSpan = $( '<span>' ).addClass( 'om-btn-bg om-btn-left' ),
270         	    centerSpan = $( '<span>' ).addClass( 'om-btn-bg om-btn-center' ),
271         	    rightSpan = $( '<span>' ).addClass( 'om-btn-bg om-btn-right' );
272         	if( options.width > 10 ){
273         		centerSpan.css( 'width',parseInt( options.width )-10 );
274         	}
275         	var buttonElement = this.buttonElement.addClass( 'om-btn-txt' );
276         	if( this.type == 'a' || this.type == 'button' ){
277         	    buttonElement.html( label );
278         	}else{
279         	    buttonElement.val( label );
280         	}
281         	if( options.icons.left ){
282         	    if( label ){
283         	        buttonElement.addClass( 'om-btn-icon' ).css( 'background-image','url('+options.icons.left+')' );
284         	    }else{
285         	        buttonElement.addClass( 'om-btn-only-icon' ).css('background-image','url('+options.icons.left+')' );
286         	    }
287         	}
288         	buttonElement.css( {'background-position':'left center','background-repeat':'no-repeat'} );
289         	buttonElement.wrap( wrapperSpan );
290         	leftSpan.insertBefore( buttonElement );
291         	rightSpan.insertAfter( buttonElement );
292         	buttonElement.wrap( centerSpan );
293         	if( options.icons.right ){
294         	    if( label != '' ){
295         	        $( '<img>' ).attr( 'src',options.icons.right ).css( {'vertical-align':'text-top','padding-left':'3px'} ).insertAfter( buttonElement );
296         	    }else{
297         	        $( '<img>' ).attr( 'src',options.icons.right ).css( 'vertical-align','text-top' ).insertAfter( buttonElement );
298         	    }
299             }
300         },
301         _getLabel : function(){
302         	return this.options.label || this.buttonElement.html() || this.buttonElement.text() || this.buttonElement.val();
303         },
304         _determineButtonType: function() {
305     		if ( this.element.is("input") ) {
306     			this.type = "input";
307     		}  else if ( this.element.is("a") ) {
308     			this.type = "a";
309     		} else if ( this.element.is('button') ){
310     			this.type = "button";
311     		}
312     		this.buttonElement = this.element;
313     	}
314     });
315 })(jQuery);
