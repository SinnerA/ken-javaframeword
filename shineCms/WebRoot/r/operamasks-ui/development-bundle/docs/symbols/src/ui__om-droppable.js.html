 42 $.widget("om.omDroppable", {

 44 	options: {
 45 		/**
 46          * 指定可以接受的拖动元素，默认任意元素都可以被接受。

 58          * @type String

 75          * @name omDroppable#hoverClass

 77          * @default 无
 78          * @example
 79          * $("#selector").omDroppable({hoverClass:"om-state-hover"});

 84 		 * 设置放置功能是否可用。

 96          * @type Function
 97          * @example

106          * @example

112          * @event
113          * @name omDroppable#onDragOut
114          * @type Function
115          * @example
116          *   $("#selector").omDroppable({onDragOut : function(event, source) {doSomething...}});

128 	_create: function() {
129 
130 		var o = this.options, accept = o.accept;
131 		this.isover = 0; this.isout = 1;
132 
133 		this.accept = $.isFunction(accept) ? accept : function(d) {
134 			return d.is(accept);
135 		};
136 
137 		//Store the droppable's proportions
138 		this.proportions = { width: this.element[0].offsetWidth, height: this.element[0].offsetHeight };
139 
140 		// Add the reference and positions to the manager
141 		$.ui.ddmanager.droppables[o._scope] = $.ui.ddmanager.droppables[o._scope] || [];
142 		$.ui.ddmanager.droppables[o._scope].push(this);
143 
144 		this.element.addClass("om-droppable");

167      * 删除元素的放置功能.

180 
181 		this.element
182 			.removeClass("om-droppable om-droppable-disabled")
183 			.removeData("omDroppable")

200 		var draggable = $.ui.ddmanager.current;
201 		if(this.options.activeClass) this.element.addClass(this.options.activeClass);
202 		(draggable && this._trigger('onDragStart', event, draggable.currentItem || draggable.element));

206 		var draggable = $.ui.ddmanager.current;
207 		if(this.options.activeClass) this.element.removeClass(this.options.activeClass);
208 		//(draggable && this._trigger('onDeactivate', event, draggable.currentItem || draggable.element));
209 	},

211 	_over: function(event) {
212 
213 		var draggable = $.ui.ddmanager.current;
214 		if (!draggable || (draggable.currentItem || draggable.element)[0] == this.element[0]) return; // Bail if draggable and droppable are same element
215 
216 		if (this.accept.call(this.element[0],(draggable.currentItem || draggable.element))) {
217 			if(this.options.hoverClass) this.element.addClass(this.options.hoverClass);
218 			this._trigger('onDragOver', event, draggable.currentItem || draggable.element);
219 		}
220 
221 	},
222 
223 	_out: function(event) {
224 
225 		var draggable = $.ui.ddmanager.current;
226 		if (!draggable || (draggable.currentItem || draggable.element)[0] == this.element[0]) return; // Bail if draggable and droppable are same element
227 
228 		if (this.accept.call(this.element[0],(draggable.currentItem || draggable.element))) {
229 			if(this.options.hoverClass) this.element.removeClass(this.options.hoverClass);
230 			this._trigger('onDragOut', event, draggable.currentItem || draggable.element);
231 		}
232 
233 	},
234 
235 	_drop: function(event,custom) {
236 
237 		var draggable = custom || $.ui.ddmanager.current;
238 		if (!draggable || (draggable.currentItem || draggable.element)[0] == this.element[0]) return false; // Bail if draggable and droppable are same element
239 
240 		var childrenIntersection = false;
241 		this.element.find(":data(omDroppable)").not(".om-draggable-dragging").each(function() {
242 			var inst = $.data(this, 'omDroppable');
243 			if(
244 				inst.options.greedy
245 				&& !inst.options.disabled
246 				&& inst.options._scope == draggable.options._scope
247 				&& inst.accept.call(inst.element[0], (draggable.currentItem || draggable.element))
248 				&& $.ui.intersect(draggable, $.extend(inst, { offset: inst.element.offset() }))
249 			) { childrenIntersection = true; return false; }
250 		});
251 		if(childrenIntersection) return false;
252 
253 		if(this.accept.call(this.element[0],(draggable.currentItem || draggable.element))) {
254 			if(this.options.activeClass) this.element.removeClass(this.options.activeClass);
255 			if(this.options.hoverClass) this.element.removeClass(this.options.hoverClass);
256 			this._trigger('onDrop', event, draggable.currentItem || draggable.element);
257 			return this.element;
258 		}
259 
260 		return false;
261 
262 	}

267 $.ui.intersect = function(draggable, droppable) {
268 
269 	if (!droppable.offset) return false;
270 
271 	var x1 = (draggable.positionAbs || draggable.position.absolute).left, x2 = x1 + draggable.helperProportions.width,
272 		y1 = (draggable.positionAbs || draggable.position.absolute).top, y2 = y1 + draggable.helperProportions.height;
273 	var l = droppable.offset.left, r = l + droppable.proportions.width,
274 		t = droppable.offset.top, b = t + droppable.proportions.height;
275 	return (l < x1 + (draggable.helperProportions.width / 2) // Right Half
276 		&& x2 - (draggable.helperProportions.width / 2) < r // Left Half
277 		&& t < y1 + (draggable.helperProportions.height / 2) // Bottom Half
278 		&& y2 - (draggable.helperProportions.height / 2) < b ); // Top Half
279 };

281 /*
282 	This manager tracks offsets of draggables and droppables
283 */
284 $.ui.ddmanager = {
285 	current: null,
286 	droppables: { 'default': [] },
287 	prepareOffsets: function(t, event) {
288 
289 		var m = $.ui.ddmanager.droppables[t.options._scope] || [];
290 		var type = event ? event.type : null; // workaround for #2317
291 		var list = (t.currentItem || t.element).find(":data(omDroppable)").andSelf();
292 
293 		droppablesLoop: for (var i = 0; i < m.length; i++) {
294 
295 			if(m[i].options.disabled || (t && !m[i].accept.call(m[i].element[0],(t.currentItem || t.element)))) continue;	//No disabled and non-accepted
296 			for (var j=0; j < list.length; j++) { if(list[j] == m[i].element[0]) { m[i].proportions.height = 0; continue droppablesLoop; } }; //Filter out elements in the current dragged item
297 			m[i].visible = m[i].element.css("display") != "none"; if(!m[i].visible) continue; 									//If the element is not visible, continue
298 
299 			if(type == "mousedown") m[i]._activate.call(m[i], event); //Activate the droppable if used directly from draggables
300 
301 			m[i].offset = m[i].element.offset();
302 			m[i].proportions = { width: m[i].element[0].offsetWidth, height: m[i].element[0].offsetHeight };
303 
304 		}
305 
306 	},
307 	drop: function(draggable, event) {
308 
309 		var dropped = false;
310 		$.each($.ui.ddmanager.droppables[draggable.options._scope] || [], function() {
311 
312 			if(!this.options) return;
313 			if (!this.options.disabled && this.visible && $.ui.intersect(draggable, this))
314 				dropped = dropped || this._drop.call(this, event);
315 
316 			if (!this.options.disabled && this.visible && this.accept.call(this.element[0],(draggable.currentItem || draggable.element))) {
317 				this.isout = 1; this.isover = 0;
318 				this._deactivate.call(this, event);
319 			}
320 
321 		});
322 		return dropped;
323 
324 	},
325 	dragStart: function( draggable, event ) {
326 		//Listen for scrolling so that if the dragging causes scrolling the position of the droppables can be recalculated (see #5003)
327 		draggable.element.parentsUntil( "body" ).bind( "scroll.droppable", function() {
328 			if( !draggable.options.refreshPositions ) $.ui.ddmanager.prepareOffsets( draggable, event );
329 		});
330 	},
331 	drag: function(draggable, event) {
332 
333 		//If you have a highly dynamic page, you might try this option. It renders positions every time you move the mouse.
334 		if(draggable.options.refreshPositions) $.ui.ddmanager.prepareOffsets(draggable, event);
335 
336 		//Run through all droppables and check their positions based on specific tolerance options
337 		$.each($.ui.ddmanager.droppables[draggable.options._scope] || [], function() {
338 
339 			if(this.options.disabled || this.greedyChild || !this.visible) return;
340 			var intersects = $.ui.intersect(draggable, this);
341 
342 			var c = !intersects && this.isover == 1 ? 'isout' : (intersects && this.isover == 0 ? 'isover' : null);
343 			if(!c) return;
344 
345 			var parentInstance;
346 			if (this.options.greedy) {
347 				var parent = this.element.parents(':data(omDroppable):eq(0)');
348 				if (parent.length) {
349 					parentInstance = $.data(parent[0], 'omDroppable');
350 					parentInstance.greedyChild = (c == 'isover' ? 1 : 0);
351 				}
352 			}
353 
354 			// we just moved into a greedy child
355 			if (parentInstance && c == 'isover') {
356 				parentInstance['isover'] = 0;
357 				parentInstance['isout'] = 1;
358 				parentInstance._out.call(parentInstance, event);
359 			}
360 
361 			this[c] = 1; this[c == 'isout' ? 'isover' : 'isout'] = 0;
362 			this[c == "isover" ? "_over" : "_out"].call(this, event);
363 
364 			// we just moved out of a greedy child
365 			if (parentInstance && c == 'isout') {
366 				parentInstance['isout'] = 0;
367 				parentInstance['isover'] = 1;
368 				parentInstance._over.call(parentInstance, event);
369 			}
370 		});
371 
372 	},
373 	dragStop: function( draggable, event ) {
374 		draggable.element.parentsUntil( "body" ).unbind( "scroll.droppable" );
375 		//Call prepareOffsets one final time since IE does not fire return scroll events when overflow was caused by drag (see #5003)
376 		if( !draggable.options.refreshPositions ) $.ui.ddmanager.prepareOffsets( draggable, event );
377 	}
378 };
